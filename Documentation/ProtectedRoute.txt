// src/components/ProtectedRoute.tsx

import React from 'react';
import { Navigate } from 'react-router-dom';
import { useAuth } from '../hooks/useAuth';
import { ChildrenProps } from '../types'; // Importamos nosso novo tipo

const ProtectedRoute: React.FC<ChildrenProps> = ({ children }) => {
  const { isLoggedIn, isLoading } = useAuth();

  console.log(`ProtectedRoute: Status - isLoading=${isLoading}, isLoggedIn=${isLoggedIn}`);

  if (isLoading) {
    console.log("ProtectedRoute: Ainda carregando, aguardando...");
    return <div className="text-white text-center pt-32">Verificando acesso...</div>;
  }

  if (!isLoggedIn) {
    console.log("ProtectedRoute: Não está logado, redirecionando para /entrar");
    return <Navigate to="/entrar" replace />;
  }

  console.log("ProtectedRoute: Acesso permitido, renderizando página.");
  return <>{children}</>;
};

export default ProtectedRoute;
